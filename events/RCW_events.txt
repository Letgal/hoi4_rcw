add_namespace = rcw

country_event = {
	id = rcw.1
	title = rcw.1.t
	desc = rcw.1.d
	picture = GFX_report_event_generic_riot

	fire_only_once = yes

	trigger = {
		NOT = { has_global_flag = RCW_ongoing }
		tag = RUS
		has_government = communism
		date > 1917.11.11
	}

	mean_time_to_happen = { days = 7 }

	option = {
		name = rcw.1.a
		start_civil_war = {
			ruling_party = ROOT
			ideology = neutrality
			size = 0.6
			capital = 219
			states = all
			states_filter = {
				NOT = { region = 132 }
			}
			PREV = {
				RUS_provisional_government = { set_nationality = PREV.PREV }
			}
			hidden_effect = {
				set_cosmetic_tag = RUS_white
				set_party_name = {
					ideology = neutrality
					name = RUS_white_movement
					long_name = RUS_white_movement
				}
			}
			add_country_leader_role = {
				character = RUS_provisional_government
				promote_leader = yes
				country_leader = {
					ideology = despotism
					expire = "1953.3.1.1"
					traits = {
						emigre_connections
						desperate_gambit
						dysfunctional_assembly
					}
				}
			}
			add_ideas = WHI_factionalism
			PREV = {
				if = {
					limit = { has_war_with = GER }
					set_truce = {
						target = GER
						days = 60
					}
				}
				release = FIN
				release = MOL
			}
		}
	}
}

country_event = {
	id = rcw.2
	title = rcw.2.t
	desc = rcw.2.d
	picture = GFX_report_event_generic_riot

	fire_only_once = yes

	trigger = {
		has_global_flag = RCW_ongoing
		OR = {
			original_tag = RUS
			original_tag = SOV
		}
		UKR = {
			exists = no
			NOT = {
				has_war_with_wargoal_against = {
					target = RUS
					type = independence_wargoal
				}
			}
		}
	}

	mean_time_to_happen = { days = 60 }

	immediate = {
		hidden_effect = {
			202 = { transfer_state_to = UKR }
			203 = { transfer_state_to = UKR }
		}
	}

	option = {
		name = rcw.2.a
		hidden_effect = {
			UKR = {
				declare_war_on = {
					target = RUS
					type = independence_wargoal
				}
				declare_war_on = {
					target = SOV
					type = independence_wargoal
				}
			}
		}
		UKR = { custom_effect_tooltip = country_declares_independence_capital }
	}
	option = {
		name = rcw.2.b
		hidden_effect = {
			UKR = {
				declare_war_on = {
					target = RUS
					type = independence_wargoal
				}
				declare_war_on = {
					target = SOV
					type = independence_wargoal
				}
			}
		}
		UKR = {
			custom_effect_tooltip = country_declares_independence_capital
			change_tag_from = ROOT
		}
	}
}
